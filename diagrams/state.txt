        direction LR

        AnalyzingGraph --> AtomicRecovery : Soft Conflict
        AnalyzingGraph --> SessionRollback : Logic Inconsistency
        
        AtomicRecovery --> Resolving : Map existing answers\nto new DAG
        SessionRollback --> Resolving : Revert to last\nvalid stable node
        
        Resolving --> [*]
    }
    
    %% Return to session or finish
    ConflictState --> ActiveSession : UI State Restored
